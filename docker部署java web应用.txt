docker部署java web应用

1. docker pull daocloud.io/library/tomcat:8.5-jre8-slim

	tomcat镜像自带jdk

2. 将war包放在工作目录中

3. vi Dockerfile

	from daocloud.io/library/tomcat:8.5-jre8-slim

	MAINTAINER yeta 429721783@qq.com

	COPY docker1.war /usr/local/tomcat/webapps
	
4. docker build -t docker1:latest .

5. docker run -d -p 8080:8080 docker1
	
	docker exec -it 75737e85ce3f /bin/bash		进入容器内部

6. docker pull daocloud.io/library/mysql:5.7.7

7. docker run -d -p 3307:3306 -e MYSQL_ROOT_PASSWORD=root -e MYSQL_DATABASE=docker daocloud.io/library/mysql:5.7.7
	
	netstat -na|grep 3306
